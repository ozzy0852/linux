ls - листинг каталогов/файлов

pwd (print working dir) - показать тек/каталог

cd (change dir) - смена каталога

(без аргументов - вернуться в дом/каталог)

mkdir (make dir) - создать каталог

rm (remove) - удалить файл/каталог

mv (move) - переместить/переименовать

cp (copy) - копировать

cat - вывод содержимого файла в терминал

id - вывод информации о пользователе

uname - вывод информации о системе
==
cd (перейти в дом/каталог)
mkdir backups (создать каталог)

~ (дом/каталог)
. (тек/каталог)
.. (каталог выше)

root - название пользователя по умолчанию
(главного администратора компьютера)
